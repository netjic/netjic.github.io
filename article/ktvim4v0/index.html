vueMapGetters原理

约 418 字大约 1 分钟

vuevueX

2025-01-10

1. 基本原理

// Vuex内部实现mapGetters的简化版本
function mapGetters(getters) {
    const res = {}
    
    // 处理数组形式
    if (Array.isArray(getters)) {
        getters.forEach(key => {
            // 生成同名计算属性
            res[key] = function() {
                return this.$store.getters[key]
            }
        })
    } 
    // 处理对象形式
    else {
        Object.keys(getters).forEach(alias => {
            // 生成别名计算属性
            res[alias] = function() {
                return this.$store.getters[getters[alias]]
            }
        })
    }
    
    return res
}

2. 使用示例

// Vuex store
const store = createStore({
    state: {
        todos: [
            { id: 1, text: '学习Vue', done: true },
            { id: 2, text: '学习Vuex', done: false }
        ]
    },
    getters: {
        doneTodos: state => state.todos.filter(todo => todo.done)
    }
})

// 组件中使用
export default {
    computed: {
        // 1. 数组形式：保持原名
        ...mapGetters(['doneTodos']),
        
        // 2. 对象形式：使用别名
        ...mapGetters({
            completedTodos: 'doneTodos'
        })
    }
}

3. 展开后的实际代码

// 数组形式展开后等价于：
computed: {
    doneTodos() {
        return this.$store.getters.doneTodos
    }
}

// 对象形式展开后等价于：
computed: {
    completedTodos() {
        return this.$store.getters.doneTodos
    }
}

4. 混合使用示例

export default {
    computed: {
        // 本地计算属性
        localComputed() {
            return this.someData
        },
        
        // 映射的getters
        ...mapGetters([
            'doneTodos',
            'pendingTodos'
        ]),
        
        // 带别名的getters
        ...mapGetters({
            completed: 'doneTodos',
            pending: 'pendingTodos'
        })
    }
}

5. 实际应用场景

// store.js
const store = createStore({
    state: {
        products: [
            { id: 1, name: 'iPhone', price: 999, inStock: true },
            { id: 2, name: 'iPad', price: 799, inStock: false }
        ]
    },
    getters: {
        availableProducts: state => 
            state.products.filter(p => p.inStock),
        productCount: state => state.products.length
    }
})

// ProductList.vue
export default {
    computed: {
        // 方式1：数组形式
        ...mapGetters([
            'availableProducts',
            'productCount'
        ]),
        
        // 方式2：对象形式（重命名）
        ...mapGetters({
            inStock: 'availableProducts',
            total: 'productCount'
        })
    }
}

这个语法的主要优点是：

  1. 简化了Vuex getter的映射过程

  2. 提供了灵活的命名选项

  3. 减少了重复代码

  4. 使组件代码更清晰

  5. 便于维护和重构

mapGetters是Vuex提供的辅助函数之一，它通过生成计算属性的方式，简化了在组件中访问store getters的过程。
