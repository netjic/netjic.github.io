1. CommonJS

2009年创建，最初用于浏览器之外的JavaScript环境，Node.js采用并推广，成为Node.js默认模块系统，解决了JavaScript没有模块化的问题。 特点：同步加载（适合服务器环境）、动态加载（运行时加载）、模块缓存、模块作用域

1.1.基本导出方式

1.1.1 使用 module.exports 导出对象

// user.js
// user.js
module.exports = {
    name: 'Tom',
    age: 18,
    sayHi() {
        console.log('Hi');
    }
};

// 导入
const user = require('./user');
console.log(user.name);    // Tom
user.sayHi();              // Hi

1.1.2 使用 exports 逐个导出

// math.js
exports.add = (a, b) => a + b;
exports.subtract = (a, b) => a - b;

// 导入
const math = require('./math');
console.log(math.add(2, 3));    // 5

// 解构导入
const { add, subtract } = require('./math');            // Hi

1.2.导出单个值

1.2.1 导出类

// Person.js
module.exports = class Person {
    constructor(name) {
        this.name = name;
    }
};

1.2.2导出函数

// logger.js
module.exports = function log(msg) {
    console.log(msg);
};

// 导入
const logger = require('./logger');
logger('Hello');

1.3.混合导出

1.3.1 主导出附加属性

// db.js
class Database {
    constructor() {
        this.data = [];
    }
}

// 主导出
module.exports = Database;
// 添加额外属性
module.exports.VERSION = '1.0.0';

// 导入
const Database = require('./db');
const db = new Database();
console.log(Database.VERSION);  // 1.0.0

1.4.特殊导入

1.4.1 导入 JSON

// config.json
{
    "port": 3000
}

// 导入 JSON
const config = require('./config.json');
console.log(config.port);  // 3000

1.4.2 导入核心模块

const fs = require('fs');
fs.readFile('./file.txt', (err, data) => {
    if (err) throw err;
    console.log(data);
});

1.4.3 导入第三方模块

const express = require('express');
const app = express();
app.get('/', (req, res) => {
    res.send('Hello World');
});
app.listen(3000, () => {
    console.log('Server is running on port 3000');
});

1.5.条件导出

// db.js
if (process.env.NODE_ENV === 'production') {
    module.exports = require('./prod-db');
} else {
    module.exports = require('./dev-db');
}

1.6.注意事项

// ❌ 错误方式：直接赋值 exports
exports = { name: 'Tom' };  // 这样会断开与 module.exports 的引用

// ✅ 正确方式
module.exports = { name: 'Tom' };
// 或者
exports.name = 'Tom';

1.7.错误处理

// 安全的导入方式
try {
    const myModule = require('./myModule');
} catch (err) {
    console.error('模块加载失败:', err);
}

1.8.模块缓存

// 模块在第一次加载后会被缓存
const user1 = require('./user'); // 模块被加载并执行
const user2 = require('./user'); // 直接返回缓存的结果，不会重新执行模块
console.log(user1 === user2);  // true


user1.increment();
console.log(user2.count);  // 输出: 1，因为 user1 和 user2 指向同一个对象

// 清除缓存
delete require.cache[require.resolve('./user')];

// 删除后重新加载
const user3 = require('./user');  // 模块会重新执行
console.log(user3.count);  // 输出: 0，因为 user3 是新的实例

1.9.注意事项总结

  • exports 是 module.exports 的引用
  • 不能直接给 exports 赋值
  • 模块路径区分大小写
  • 模块在第一次加载后会被缓存
  • JSON 文件会自动解析
  • 可以省略 .js 扩展名

2. ES6 Module

2015年随ES6标准发布，JavaScript官方模块化规范，设计目标是同时支持浏览器和服务器。 特点：异步加载（适合浏览器环境）、静态加载（编译时加载）、模块缓存、模块作用域

2.1 命名导出 (Named Exports)

2.1.1 单个导出

// math.js
export const add = (a, b) => a + b;
export const subtract = (a, b) => a - b;

// 导入方式
import { add, subtract } from './math.js';
add(1, 2);  // 3

2.1.2 批量导出

// utils.js
const multiply = (a, b) => a * b;
const divide = (a, b) => a / b;
export { multiply, divide };

// 导入方式
import { multiply, divide } from './utils.js';
// 或者重命名导入
import { multiply as mul, divide as div } from './utils.js';

2.1.3 全部导入

// math.js
export const add = (a, b) => a + b;
export const subtract = (a, b) => a - b;

// 导入方式
import * as mathUtils from './math.js';
mathUtils.add(1, 2);      // 3
mathUtils.subtract(5, 3); // 2

2.2 默认导出 (Default Export)

2.2.1 导出类

// User.js
export default class User {
    constructor(name) {
        this.name = name;
    }
}

// 导入方式
import User from './User.js';
const user = new User('Tom');
console.log(user.name);  // Tom

2.2.2 导出对象

// config.js
export default {
    apiUrl: 'https://api.example.com',
    timeout: 3000
};

// 导入方式
import config from './config.js';
console.log(config.apiUrl);

2.3 混合导出

2.3.1 默认导出和命名导出混用

// api.js
export default class API {
    constructor() {
        this.baseUrl = 'https://api.example.com';
    }
}
export const VERSION = '1.0.0';
export const helper = {
    format(data) { return data; }
};

// 导入方式
import API, { VERSION, helper } from './api.js';
const api = new API();
console.log(VERSION);
helper.format({});

2.4 重新导出

2.4.1 重新导出其他模块

// index.js
export { default as User } from './User.js';
export { add, subtract } from './math.js';
export * from './utils.js';

// 导入方式
import { User, add, subtract } from './index.js';

2.5 动态导入

2.5.1 按需导入

// app.js
const loadModule = async () => {
    const module = await import('./dynamic-module.js');
    module.doSomething();
};

// 使用方式
loadModule().then(() => {
    console.log('Module loaded');
});

2.6 注意事项

2.6.1 路径说明

// 相对路径导入
import './styles.js';           // 同级目录
import '../utils/helper.js';    // 上级目录
import './services/api.js';     // 子目录

// 绝对路径导入（基于项目配置）
import '@/components/Button';   // 需要配置别名

2.6.2 文件扩展名

// 在支持 ES Modules 的环境中通常需要添加 .js 扩展名
import './module.js';

// 在某些构建工具中可以省略扩展名
import './module';

3.CommonJS 和 ES Modules 对比总结

3.1.语法差异

CommonJS

// 导出
module.exports = { ... }
exports.foo = ... 

// 导入
const module = require('./module')

ES Modules

// 导出
export default { ... }
export const foo = ...

// 导入
import module from './module'
import { foo } from './module'

3.2.主要区别

加载时机

  • CommonJS: 同步加载，运行时加载
  • ES Modules: 异步加载，编译时加载（静态分析） 缓存机制
  • CommonJS: 缓存模块实例
  • ES Modules: 缓存模块引用 模块查找
  • CommonJS: node_modules 逐级向上查找
  • ES Modules: 需要完整路径或配置别名 使用场景
  • CommonJS: Node.js 服务端
  • ES Modules: 现代浏览器、前端工程化

3.3.特性对比

特性CommonJSES Modules
动态导入require()import()
导入时机运行时编译时
导入内容值的拷贝值的引用
模块路径可以是变量必须是字符串字面量
默认导出module.exportsexport default
命名导出exports.xxxexport const xxx

3.4.使用建议

  1. 新项目：
  • 优先使用 ES Modules
  • 更好的静态分析
  • 支持 Tree Shaking
  1. Node.js项目：
  • 可以使用 CommonJS
  • 或在 package.json 中设置 "type": "module"
  1. 兼容性考虑：
  • 需要兼容旧环境用 CommonJS
  • 现代环境用 ES Modules
  1. 混合使用：
  • 尽量避免在同一项目中混用
  • 必要时使用构建工具转换
