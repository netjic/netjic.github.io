Vuex.Store 的核心属性

约 561 字大约 2 分钟

2025-01-10

1.基本结构

const store = new Vuex.Store({
    state: {},        // 状态
    getters: {},      // 获取器
    mutations: {},    // 修改器
    actions: {},      // 动作
    modules: {}       // 模块
});

2.State（状态）

const store = new Vuex.Store({
    state: {
        count: 0,
        todos: [],
        user: {
            name: '',
            isLoggedIn: false
        }
    }
});

// 访问方式
store.state.count
// 在组件中
this.$store.state.count
// 使用mapState
...mapState(['count'])

3.Getters（获取器）

const store = new Vuex.Store({
    state: {
        todos: [
            { id: 1, text: '...', done: true },
            { id: 2, text: '...', done: false }
        ]
    },
    getters: {
        // 基本getter
        doneTodos: state => {
            return state.todos.filter(todo => todo.done)
        },
        // 接收其他getter的getter
        doneTodosCount: (state, getters) => {
            return getters.doneTodos.length
        },
        // 返回函数的getter
        getTodoById: state => id => {
            return state.todos.find(todo => todo.id === id)
        }
    }
});

// 访问方式
store.getters.doneTodos
// 在组件中
this.$store.getters.doneTodos
// 使用mapGetters
...mapGetters(['doneTodos'])

4.Mutations（修改器）

const store = new Vuex.Store({
    state: {
        count: 0
    },
    mutations: {
        // 基本mutation
        increment(state) {
            state.count++
        },
        // 带参数的mutation
        incrementBy(state, payload) {
            state.count += payload.amount
        },
        // 对象风格的mutation
        setUser(state, { name, age }) {
            state.user.name = name
            state.user.age = age
        }
    }
});

// 调用方式
store.commit('increment')
store.commit('incrementBy', { amount: 10 })
// 在组件中
this.$store.commit('increment')
// 使用mapMutations
...mapMutations(['increment'])

4.1.Mutaion的原理实现

// 简化版的Vuex Mutation实现
class Store {
    constructor(options) {
        this.state = Vue.observable(options.state)
        this._mutations = options.mutations
    }
    
    commit(type, payload) {
        // 1. 获取对应的mutation处理函数
        const mutation = this._mutations[type]
        
        if (!mutation) {
            console.error(`Unknown mutation type: ${type}`)
            return
        }
        
        // 2. 调用mutation并传入state和payload
        mutation(this.state, payload)
    }
}

4.2.为什么修改state状态要使用mutaion

  1. 状态追踪
  2. 集中管理
  3. 可预测的状态变化

5.Actions（动作）

const store = new Vuex.Store({
    state: {
        posts: []
    },
    mutations: {
        setPosts(state, posts) {
            state.posts = posts
        }
    },
    actions: {
        // 基本action
        async fetchPosts({ commit }) {
            const posts = await api.getPosts()
            commit('setPosts', posts)
        },
        // 带参数的action
        async fetchPostById({ commit }, postId) {
            const post = await api.getPost(postId)
            commit('setCurrentPost', post)

        // 组合action
        async initializeApp({ dispatch }) {

// 调用方式

// 在组件中
this.$store.dispatch('fetchPosts')
// 使用mapActions
...mapActions(['fetchPosts'])

this.$store.dispatch('fetchPosts') 原理

class Store {
    constructor(options) {
        this.state = Vue.observable(options.state)
        this._actions = options.actions
        this._mutations = options.mutations
        
        // 绑定dispatch方法的this
        this.dispatch = this.dispatch.bind(this)
    }
    
    dispatch(type, payload) {
        // 1. 获取对应的action
        const action = this._actions[type]
        
        if (!action) {
            console.error(`[vuex] unknown action type: ${type}`)
            return
        }
        
        // 2. 调用action
        return action({
            dispatch: this.dispatch,
            commit: this.commit,
            state: this.state,
            getters: this.getters
        }, payload)
    }
}

6.Modules（模块）

const moduleA = {
    namespaced: true,
    state: { ... },
    getters: { ... },
    mutations: { ... },
    actions: { ... }
};

const moduleB = {
    namespaced: true,
    state: { ... },
    modules: {
        subModule: { ... }
    }
};

const store = new Vuex.Store({
    modules: {
        a: moduleA,
        b: moduleB
    }
});

// 访问方式
store.state.a // moduleA的状态
store.state.b // moduleB的状态
// 在组件中使用命名空间模块
...mapState('a', ['someState'])
...mapGetters('a', ['someGetter'])
