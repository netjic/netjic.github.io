Redisson基本数据类型存储

约 223 字小于 1 分钟

redis

2023-11-28

项目依赖


<dependency>
    <groupId>org.redisson</groupId>
    <artifactId>redisson</artifactId>
    <version>3.10.2</version>
</dependency>

配置


@Configuration
public class RedissonConfig {
    /**
     * 配置一个临时的对象到spring容器中
     *
     * @return 一个RedissonClient的实现
     */
    @Bean
    public RedissonClient redissonClient() {
        Config config = new Config();
        config.useSingleServer()
                .setAddress("redis://127.0.0.1:6379");
        return Redisson.create(config);
    }
}

代码示例


@RestController
public class RedissonDataController {
    @Autowired
    public RedissonClient redissonClient;

    // String类型
    @GetMapping("/stringData")
    public String stringData() {
        RBucket<Object> stringData = redissonClient.getBucket("stringData");
        stringData.set("stringData");
        return (String) redissonClient.getBucket("stringData").get();
    }

    // List类型
    @GetMapping("/listData")
    public List listData() {
        redissonClient.getList("listData").addAll(List.of("listData1"));
        return redissonClient.getList("listData");
    }

    // Hash类型
    @GetMapping("/hashData")
    public String hashData() {
        redissonClient.getMap("hashData").put("hashDataKey", "hashDataValue");
        return (String) redissonClient.getMap("hashData").get("hashDataKey");
    }

    // Set类型
    @GetMapping("/setData")
    public Set setData() {
        redissonClient.getSet("setData").addAll(Set.of("set"));
        return redissonClient.getSet("setData");
    }

    // SortedSet类型
    @GetMapping("/sortedSet")
    public Set sortedSet() {
        RSortedSet<Integer> setSortSet = redissonClient.getSortedSet("setSortSet");
        setSortSet.add(7);
        setSortSet.add(6);
        return redissonClient.getSortedSet("setSortSet");
    }

    // AtomicLong类型 分布式自增ID
    @GetMapping("/incrementId")
    public Long incrementId() {
        RAtomicLong id = redissonClient.getAtomicLong("id");
        id.incrementAndGet();
        return redissonClient.getAtomicLong("id").get();
    }
}
